Creating a feed

Example http://api.cosm.com/v2/feeds
Formats xml, json
Methods POST
Returns URL of the newly created feed in headers

New environments can be created by sending a correctly formatted document to this URL or by including the data in the POST body.

An example using curl is:

curl --request POST --data-binary @datafile.txt --header "X-ApiKey: YOUR_API_KEY_HERE" http://api.cosm.com/v2/feeds

In this example you would put your document in the file "datafile.txt".

Here's an example creation request you can try using hurl.it. You'll need to supply your own API key for this to work.

The URL of the newly created feed is returned using the "Location" header.

Examples

JSON

{
  "title":"Cosm Office environment",
  "website":"http://www.example.com/",
  "version":"1.0.0",
  "tags":[
      "Tag1",
      "Tag2"
  ],
  "location":{
    "disposition":"fixed",
    "ele":"23.0",
    "name":"office",
    "lat":51.5235375648154,
    "exposure":"indoor",
    "lon":-0.0807666778564453,
    "domain":"physical"
  },
  "datastreams":[
    {
      "current_value":"123",
      "max_value":"10000.0",
      "min_value":"-10.0",
      "tags":["humidity"],
      "id":"0"
    },
    {
      "current_value":"987",
      "max_value":"10000.0",
      "min_value":"-10.0",
      "tags":["humidity"],
      "id":"1"
    }
  ]
}

A minimal version of this, with only the required parameters is:

{
  "title":"Cosm Office environment",
  "version":"1.0.0",
  "datastreams":[
    {
      "id":"0"
    },
    {
      "id":"1"
    }
  ]
}

XML

<eeml xmlns="http://www.eeml.org/xsd/0.5.1" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" version="0.5.1" xsi:schemaLocation="http://www.eeml.org/xsd/0.5.1 http://www.eeml.org/xsd/0.5.1/0.5.1.xsd">
  <environment creator="https://cosm.com/users/hdr">
    <title>Cosm Office environment</title>
    <website>http://www.example.com/</website>
    <tag>Tag1</tag>
    <tag>Tag2</tag>
    <location domain="physical" exposure="indoor" disposition="fixed">
      <name>office</name>
      <lat>51.5235375648154</lat>
      <lon>-0.0807666778564453</lon>
      <ele>23.0</ele>
    </location>
    <data id="0">
      <tag>humidity</tag>
      <min_value>0.0</min_value>
      <max_value>658.0</max_value>
      <current_value>305</current_value>
    </data>
    <data id="1">
      <tag>humidity</tag>
      <min_value>0.0</min_value>
      <max_value>658.0</max_value>
      <current_value>305</current_value>
    </data>
  </environment>
</eeml>

A minimal version of this, with only the required parameters is:

<eeml xmlns="http://www.eeml.org/xsd/0.5.1" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" version="0.5.1" xsi:schemaLocation="http://www.eeml.org/xsd/0.5.1 http://www.eeml.org/xsd/0.5.1/0.5.1.xsd">
  <environment>
    <title>Cosm Office environment</title>
    <data id="0" />
    <data id="1" />
  </environment>
</eeml>
